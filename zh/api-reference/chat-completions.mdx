## 请求体

<ParamField body="model" type="string" required>
  要使用的模型 ID（例如 `gpt-4o`、`claude-3.5-sonnet`、`gemini-2.0-flash`）。
</ParamField>

<ParamField body="messages" type="array" required>
  组成对话的消息列表。
  <Expandable title="消息对象">
    <ParamField body="role" type="string" required>
      `system`、`user`、`assistant` 或 `tool` 之一。
    </ParamField>
    <ParamField body="content" type="string" required>
      消息的内容。
    </ParamField>
    <ParamField body="name" type="string">
      参与者的可选名称。
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="stream" type="boolean" default={false}>
  如果为 `true`，返回 Server-Sent Events (SSE) 流。
</ParamField>

<ParamField body="temperature" type="number" default={1}>
  采样温度，范围 0 到 2。较低的值更具确定性。
</ParamField>

<ParamField body="max_tokens" type="integer">
  要生成的最大 token 数。
</ParamField>

<ParamField body="top_p" type="number" default={1}>
  核采样参数。
</ParamField>

<ParamField body="stop" type="string | string[]">
  停止生成的停止序列。
</ParamField>

<ParamField body="frequency_penalty" type="number" default={0}>
  token 频率惩罚（-2.0 到 2.0）。
</ParamField>

<ParamField body="presence_penalty" type="number" default={0}>
  token 存在惩罚（-2.0 到 2.0）。
</ParamField>

<ParamField body="tools" type="array">
  模型可以调用的工具列表（函数调用）。
</ParamField>

<ParamField body="tool_choice" type="string | object">
  控制调用哪个工具。`auto`、`none` 或指定的工具。
</ParamField>

<ParamField body="response_format" type="object">
  强制指定输出格式（例如 `{"type": "json_object"}`）。
</ParamField>

<ParamField body="seed" type="integer">
  用于可复现性的确定性采样种子。
</ParamField>

### OpenModex 扩展参数

<ParamField body="routing" type="object">
  为此请求配置智能路由。
  <Expandable title="路由对象">
    <ParamField body="strategy" type="string">
      `cost_optimized`、`latency_optimized` 或 `quality_optimized`。
    </ParamField>
    <ParamField body="fallback" type="string[]">
      服务端故障转移模型链。
    </ParamField>
    <ParamField body="allow_upgrade" type="boolean">
      允许路由到同一价格区间的更优模型。
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="cache" type="object">
  配置提示词缓存。
  <Expandable title="缓存对象">
    <ParamField body="enabled" type="boolean">
      为此请求启用提示词缓存。
    </ParamField>
    <ParamField body="ttl" type="integer">
      缓存生存时间，单位为秒。
    </ParamField>
  </Expandable>
</ParamField>

## 示例

<CodeGroup>
```bash cURL
curl https://api.openmodex.com/v1/chat/completions \
  -H "Authorization: Bearer omx_sk_YOUR_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "model": "gpt-4o",
    "messages": [
      {"role": "system", "content": "You are a helpful assistant."},
      {"role": "user", "content": "What is the capital of France?"}
    ],
    "temperature": 0.7,
    "max_tokens": 100
  }'
```

```typescript Node.js
import OpenModex from '@openmodex/sdk';

const client = new OpenModex({ apiKey: 'omx_sk_YOUR_KEY' });

const response = await client.chat.completions.create({
  model: 'gpt-4o',
  messages: [
    { role: 'system', content: 'You are a helpful assistant.' },
    { role: 'user', content: 'What is the capital of France?' },
  ],
  temperature: 0.7,
  max_tokens: 100,
});

console.log(response.choices[0].message.content);
```

```python Python
from openmodex import OpenModex

client = OpenModex(api_key="omx_sk_YOUR_KEY")

response = client.chat.completions.create(
    model="gpt-4o",
    messages=[
        {"role": "system", "content": "You are a helpful assistant."},
        {"role": "user", "content": "What is the capital of France?"},
    ],
    temperature=0.7,
    max_tokens=100,
)

print(response.choices[0].message.content)
```
